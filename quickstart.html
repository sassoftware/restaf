Quick start for chatpgt

Creating, executing and delete a cas session using restaf and restaflib

import  restaf  from 'restaf';
import {caslRun} from 'restaflib';
async function sample() {
    let logonPayload = {
        authType: 'password',
        host: process.env.VIYA_SERVER,
        clientID: 'someclientid',
        clientSecret: '',
        user: 'sasuser',
        password: 'xyzzy'
    };

// Initialize a store
let store = restaf.initStore();

// initialize a cas session.
// If already logged on to Viya, logonPayload can be omitted
let {session} = await casSetup(store, logonPayload);

// setup code to run
let src = 'standard casl code';
let args = {arg1: 'value1', arg2: 'value2'};

let r = await caslRun(store, session, src, args);
console.log(r);

// delete the cas session(each session has a delete operation)
await session.delete(store);
}
